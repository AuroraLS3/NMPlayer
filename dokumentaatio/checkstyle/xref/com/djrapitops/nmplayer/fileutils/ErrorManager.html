6   package com.djrapitops.nmplayer.fileutils;
7   
8   import com.djrapitops.nmplayer.messaging.MessageSender;
9   import com.djrapitops.nmplayer.messaging.Phrase;
10  import java.io.File;
11  import java.io.FileWriter;
12  import java.io.IOException;
13  import java.io.PrintWriter;
14  import java.util.Collection;
15  import java.util.Date;
16  
17  /**
18   *
19   * @author Rsl1122
20   */
21  public class ErrorManager {
22  
23      /**
24       * Logs caught Throwable to Errors.txt
25       *
26       * @param source Name of the source class
27       * @param e Thrown error
28       */
29      public static void toLog(String source, Throwable e) {
30          MessageSender.getInstance().send(Phrase.ERROR + "");
31          toLog(source + " Caught " + e);
32          for (StackTraceElement x : e.getStackTrace()) {
33              toLog("  " + x);
34          }
35          toLog("");
36      }
37  
38      /**
39       * Logs multiple caught Throwables to Errors.txt.
40       *
41       * @param source Class name the exception was caught in.
42       * @param e Collection of Throwables, eg NullPointerException
43       */
44      public static void toLog(String source, Collection<Throwable> e) {
45          for (Throwable ex : e) {
46              toLog(source, ex);
47          }
48      }
49  
50      /**
51       * Logs a message to the Errors.txt with a timestamp.
52       *
53       * @param message Message to log to Errors.txt
54       */
55      public static void toLog(String message) {
56          FileWriter fw = null;
57          PrintWriter pw = null;
58          try {
59              File log = new File("Errors.txt");
60              if (!log.exists()) {
61                  log.createNewFile();
62              }
63              fw = new FileWriter(log, true);
64              pw = new PrintWriter(fw);
65              String timestamp = formatTimeStamp(new Date().getTime());
66              pw.println("[" + timestamp + "] " + message);
67              pw.flush();
68  
69          } catch (IOException ex) {
70          } finally {
71              if (pw != null) {
72                  pw.close();
73              }
74              if (fw != null) {
75                  try {
76                      fw.close();
77                  } catch (IOException ex) {
78                  }
79              }
80          }
81      }
82  
83      /**
84       * Format a long value into human readable format of (month) (day)
85       * (hh:mm:ss).
86       *
87       * @param ms ms since Epoch Date 1970
88       * @return String of format (month) (day) (hh:mm:ss)
89       */
90      public static String formatTimeStamp(long ms) {
91          Date sfd = new Date(ms);
92          return ("" + sfd).substring(4, 19);
93      }
94  }
