85      public static void downloadUrl(String url) {
86          int i = url.indexOf("v=") + 2;
87          int x = url.indexOf("&", i);
88          if (x < 0) {
89              x = url.length();
90          }
91          download(url.substring(i, x));
92      }

94      public static void download(String id) {
95          String[] flvUrls = getFLVFormatUrls(getURLS(id));
96          if (flvUrls != null) {
97              boolean downloaded = false;
98              for (String flv : flvUrls) {
99                  int tag = getTag(flv);
100                 //Download only target to avoid issues
101                 if (tag == TARGET_QUALITY) {
102                     downloaded = downloadVideo(flv);
103                 }
104                 if (downloaded) {
105                     break;
106                 }
107             }
108         }
109     }

135     public static InputStream getVideoInfo(String id) {
136         try {
137             HttpURLConnection connection = con("http://www.youtube.com/get_video_info?video_id=" + id + "&asv=3&el=detailpage&hl=en_US");
138             connection.setRequestMethod("GET");
139             return connection.getInputStream();
140         } catch (Exception ex) {
141             handleException(ex);
142         }
143         return null;

146     private static String[] getURLS(String id) {
147         try {
148             InputStream is = getVideoInfo(id);
149             if (is == null) {
150                 return null;
151             }
152             BufferedReader reader = new BufferedReader(new InputStreamReader(is, "UTF-8"));
153             LinkedList<String> urls = new LinkedList<>();
154             String title = null;
155             for (String l; (l = reader.readLine()) != null;) {
156                 for (String p : l.split("&")) {
157                     String key = p.substring(0, p.indexOf('='));
158                     String value = p.substring(p.indexOf('=') + 1);
159                     if (key.equals("url_encoded_fmt_stream_map")) {
160                         value = decode(value);
161                         for (String u : value.split("url=")) {
162                             u = getCorrectURL(decode(u));
163                             if (!u.startsWith("http") && !isValid(u)) {
164                                 continue;
165                             }
166                             urls.add(u);
167                         }
168                     } else if (key.equals("title")) {
169                         title = value.replace("+", "%20");
170                     }
171                 }
172             }
173             if (title == null) {
174                 throw new RuntimeException("Failed to find title can't complete url");
175             }
176             String[] url_map = urls.toArray(new String[urls.size()]);
177             for (int i = 0; i < url_map.length; i++) {
178                 url_map[i] += url_map[i].endsWith("&") ? "title=" + title : "&title=" + title;
179             }
180             return url_map;
181         } catch (IOException | RuntimeException e) {
182             handleException(e);
183         }
184         return null;
185     }

192     private static HttpURLConnection con(String url) throws Exception {
193         URL u = new URL(url);
194         HttpURLConnection connection = (HttpURLConnection) u.openConnection();
195         connection.setRequestProperty("User-Agent", "Mozilla/5.0 (Windows NT 6.1) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/27.0.1453.110 Safari/537.3");
196         return connection;

199     private static String getCorrectURL(String input) {
200         StringBuilder builder = new StringBuilder(input.substring(0, input.indexOf('?') + 1));
201         String[] params = input.substring(input.indexOf('?') + 1).split("&");
202         LinkedList<String> keys = new LinkedList<>();
203         boolean first = true;
204         for (String param : params) {
205             String key = param;
206             try {
207                 key = param.substring(0, param.indexOf('='));
208             } catch (Exception ex) {
209                 System.out.println(param);
210             }
211             if (keys.contains(key) || BAD_KEYS.contains(key)) {
212                 continue;
213             }
214             keys.add(key);
215             if (key.equals("sig")) {
216                 builder.append(first ? "" : "&").append("signature=").append(param.substring(4));
217             } else {
218                 if (param.contains(",quality=")) {
219                     param = remove(param, ",quality=", "_end_");
220                 }
221                 if (param.contains(",type=")) {
222                     param = remove(param, ",type=", "_end_");
223                 }
224                 if (param.contains(",fallback_host")) {
225                     param = remove(param, ",fallback_host", ".com");
226                 }
227                 builder.append(first ? "" : "&").append(param);
228             }
229             if (first) {
230                 first = false;
231             }
232         }
233         return builder.toString();
234     }

236     private static String remove(String text, String start, String end) {
237         int l = text.indexOf(start);
238         return text.replace(text.substring(l, end.equals("_end_") ? text.length() : text.indexOf(end, l)), "");
239     }

241     public static boolean downloadVideo(String url) {
242         System.out.println("Trying: " + url);
243         FlvAudioInputStream input = null;
244         FileOutputStream output = null;
245         try {
246             String file = getSaveFilePath(url.substring(url.lastIndexOf('&') + 7));
247             if (file != null) {
248                 output = new FileOutputStream(file);
249                 HttpURLConnection con = con(url);
250                 con.connect();
251                 if (progress != null) {
252                     progress.setTotal(con.getContentLength());
253                 }
254                 input = new FlvAudioInputStream(con.getInputStream());
255                 byte[] buffer = new byte[8192];
256                 int len;
257                 while ((len = input.read(buffer, 0, 8192)) > 0) {
258                     output.write(buffer, 0, len);
259                     if (progress != null) {
260                         progress.update((int) input.getOffset());
261                     }
262                 }
263                 return true;
264             }
265         } catch (Exception ex) {
266             handleException(ex);
267             return false;
268         } finally {
269             try {
270                 if (input != null) {
271                     input.close();
272                 }
273                 if (output != null) {
274                     output.close();
275                 }
276             } catch (IOException ex) {
277                 handleException(ex);
278             }
279         }
280         return false;
281     }

292     public static String decode(String s) {
293         try {
294             return URLDecoder.decode(s, "UTF-8");
295         } catch (UnsupportedEncodingException e) {
296             ErrorManager.toLog("com.djrapitops.fileutils.downloading.YoutubeMp3Downloader", e);
297         }
298         return s;
299     }
