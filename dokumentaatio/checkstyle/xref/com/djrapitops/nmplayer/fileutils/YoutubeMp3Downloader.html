30  public class YoutubeMp3Downloader {
31  
32      public static String[] getLink(String url) throws ClientProtocolException, IOException {
33          url = url.replace("https://www.youtube.com/watch?v=", "");
34          url = url.replace("http://www.youtube.com/watch?v=", "");
35          boolean passCode = false;
36          String h = "";
37          String title = "";
38          String result = "";
39          String[] returnVal = {"", ""};
40          Map<String, String> jsonTable;
41          DefaultHttpClient httpClient = new DefaultHttpClient();
42          HttpGet httpInitialGet = new HttpGet("http://www.youtube-mp3.org/api/pushItem/?item=http%3A//www.youtube.com/watch%3Fv%3D" + url + "&xy=_");
43          httpInitialGet.addHeader("Accept-Location", "*");
44          httpInitialGet.addHeader("Referrer", "http://www.youtube-mp3.org");
45          HttpParams params = new SyncBasicHttpParams();
46          HttpProtocolParams.setVersion(params, HttpVersion.HTTP_1_1);
47          HttpProtocolParams.setUserAgent(params, "Mozilla/5.0 (Windows NT 6.1; WOW64) AppleWebKit/537.1 (KHTML, like Gecko) Chrome/22.0.1207.1 Safari/537.1");
48          httpInitialGet.setParams(params);
49          HttpResponse firstResponse = httpClient.execute(httpInitialGet);
50  
51          try {
52              if (firstResponse.getStatusLine().toString().contains("200")) {
53                  passCode = true;
54              }
55          } finally {
56              httpInitialGet.releaseConnection();
57          }
58  
59          if (passCode) {
60              while (true) {
61                  HttpGet httpStatusGet = new HttpGet("http://www.youtube-mp3.org/api/itemInfo/?video_id=" + url + "&adloc=");
62                  httpStatusGet.addHeader("Accept-Location", "*");
63                  httpStatusGet.addHeader("Referrer", "http://www.youtube-mp3.org");
64                  httpStatusGet.setParams(params);
65                  HttpResponse secondResponse = httpClient.execute(httpStatusGet);
66                  HttpEntity secondEntity = secondResponse.getEntity();
67                  InputStream is = secondEntity.getContent();
68                  try {
69                      BufferedReader reader = new BufferedReader(new InputStreamReader(is, "iso-8859-1"), 8);
70                      StringBuilder sb = new StringBuilder();
71                      List<String> lines = reader.lines().collect(Collectors.toList());
72                      lines.forEach((line) -> {
73                          sb.append(line).append("\n");
74                      });
75                      is.close();
76                      result = sb.toString();
77                  } catch (IOException e) {
78                      ErrorManager.toLog(YoutubeMp3Downloader.class.getName(), e);
79                  }
80  
81                  httpStatusGet.releaseConnection();
82                  result = result.replaceAll("\\}.*", "}");
83                  result = result.replaceAll(".*?\\{", "{");
84                  try {
85                      JSONObject jsonData = new JSONObject(result);
86                      JSONArray jsonArray = jsonData.names();
87                      JSONArray valArray = jsonData.toJSONArray(jsonArray);
88                      jsonTable = new HashMap<>(jsonArray.length());
89                      for (int i = 0; i < jsonArray.length(); i++) {
90                          jsonTable.put(jsonArray.get(i).toString(), valArray.get(i).toString());
91                      }
92                      if (jsonTable.get("status").equals("serving")) {
93                          h = jsonTable.get("h");
94                          title = jsonTable.get("title");
95                          break;
96                      }
97                  } catch (JSONException e) {
98                      ErrorManager.toLog(YoutubeMp3Downloader.class.getName(), e);
99                  }
100             }
101             returnVal[0] = "http://www.youtube-mp3.org/get?video_id=" + url + "&h=" + h;
102             returnVal[1] = title;
103             return returnVal;
104         } else {
105             ErrorManager.toLog(YoutubeMp3Downloader.class.getName(), new IllegalStateException("Video is not downloadable: " + url + " " + firstResponse.getStatusLine().toString()));
106         }
107         return null;
108     }
109 }
