35  /**
36   * JavaFx Application used as the Interface for the user.
37   *
38   * When started, initialises multiple UI components and sets them to be on a

45  public class UserInterface extends Application implements Updateable {
46  
47      private final List<Updateable> updateableComponents;
48      private Stage stage;
49      private VBox toolbar;
50  
51      /**
52       * Constructor that initializes updateableComponents List.
53       */
54      public UserInterface() {
55          updateableComponents = new ArrayList<>();
56      }
57  
58      @Override
59      public void start(Stage primaryStage) throws Exception {
60          stage = primaryStage;
61          stage.setTitle("NMPlayer");
62          stage.getIcons().add(new Image("http://djrapitops.com/uploads/NMPlayer.png"));
63          BorderPane root = new BorderPane();
64          root.setCenter(playlist());
65          toolbar = toolbar();
66          root.setBottom(toolbar);
67          root.setTop(console());
68  
69          primaryStage.setScene(new Scene(root, 400, 700));
70          primaryStage.show();
71          MusicPlayer musicPlayer = MusicPlayer.getInstance();
72          root.setOnKeyPressed((KeyEvent keyEvent) -> {
73              handleKeyPress(keyEvent);
74          });
75          try {
76              musicPlayer.init();
77          } catch (IllegalStateException e) {
78              MessageSender.getInstance().send(Phrase.ERROR_JAVAFX + "");
79          }
80          update();
81          musicPlayer.setEndOfMediaUpdate(this);
82      }
83  
84      private Node playlist() {
85          HBox changePlaylistBox = new ChangePlaylistBox(this, new AddTrackButton(this, stage));
86          VBox box = new VBox();
87          VBox playlist = new UIPlaylist(this);
88          ScrollPane scroll = new ScrollPane();
89          scroll.setContent(playlist);
90          scroll.fitToWidthProperty().set(true);
91          scroll.setHbarPolicy(ScrollBarPolicy.NEVER);
92          box.getChildren().add(changePlaylistBox);
93          box.getChildren().add(scroll);
94          updateableComponents.add((Updateable) changePlaylistBox);
95          updateableComponents.add((Updateable) playlist);
96          return box;
97      }
98  
99      private VBox console() {
100         VBox box = new VBox();
101         HBox console = new HBox();
102         TextConsole con = new TextConsole();
103         console.getChildren().add(con);
104         MessageSender.getInstance().setOutput(con);
105         box.getChildren().add(console);
106         return box;
107     }
108 
109     private VBox toolbar() {
110         VBox box = new VBox();
111         HBox controls = new HBox();
112         controls.setAlignment(Pos.CENTER_RIGHT);
113         controls.alignmentProperty().isBound();
114         controls.setSpacing(5);
115         controls.setStyle("-fx-background-color: White");
116         final PlayButton play = new PlayButton(this);
117         updateableComponents.add(play);
118         ObservableList<Node> conponents = controls.getChildren();
119         ShuffleButton shuffle = new ShuffleButton();
120         conponents.add(shuffle);
121         updateableComponents.add(shuffle);
122         conponents.add(new PreviousButton(this));
123         conponents.add(play);
124         conponents.add(new StopButton(this));
125         conponents.add(new NextButton(this));
126         VolumeSlider volumeSlider = new VolumeSlider();
127         updateableComponents.add(volumeSlider);
128         conponents.add(volumeSlider);
129         TrackProgressBar progress = new TrackProgressBar();
130         updateableComponents.add(progress);
131         box.getChildren().add(progress);
132         box.getChildren().add(controls);
133         return box;
134     }

136     @Override
137     public void update() {
138         for (Updateable u : updateableComponents) {
139             u.update();
140         }
141         MusicPlayer mp = MusicPlayer.getInstance();
142         Track currentTrack = mp.getCurrentTrack();
143         String track = "None";
144         if (currentTrack != null) {
145             track = currentTrack.toString();
146         }
147         if (toolbar != null) {
148             toolbar.requestFocus();
149         }
150         stage.setTitle((mp.isPlaying() ? "â–¶" : "") + " NMPlayer | " + TextUtils.uppercaseFirst(mp.getSelectedPlaylist()) + " | " + track);
151     }

153     /**
154      * Used for Key press functionality processing.
155      *
156      * @param keyEvent Key pressed.
157      * @throws IllegalStateException if JavaFx application not started.
158      */
159     public void handleKeyPress(KeyEvent keyEvent) throws IllegalStateException {
160         MusicPlayer musicPlayer = MusicPlayer.getInstance();
161         KeyCode key = keyEvent.getCode();
162         if (key == null) {
163             return;
164         } else {
165             switch (key) {
166                 case TRACK_NEXT:
167                 case RIGHT:
168                     musicPlayer.nextTrack();
169                     break;
170                 case TRACK_PREV:
171                 case LEFT:
172                     musicPlayer.previousTrack();
173                     break;
174                 case PLAY:
175                 case PAUSE:
176                 case SPACE:
177                     if (musicPlayer.isPlaying()) {
178                         musicPlayer.pause();
179                     } else {
180                         musicPlayer.play();
181                     }
182                     break;
183                 case STOP:
184                     musicPlayer.stop();
185                     break;
186                 case DOWN:
187                     musicPlayer.setVolume(musicPlayer.getVolume() - 0.1);
188                     break;
189                 case UP:
190                     musicPlayer.setVolume(musicPlayer.getVolume() + 0.1);
191                     break;
192                 default:
193                     return;
194             }
195         }
196         update();
197     }
