6   package com.djrapitops.nmplayer.ui.playlist;
7   
8   import com.djrapitops.nmplayer.ui.Updateable;
9   import javafx.collections.ObservableList;
10  import javafx.event.EventHandler;
11  import javafx.geometry.Pos;
12  import javafx.scene.Node;
13  import javafx.scene.control.TextField;
14  import javafx.scene.input.KeyCode;
15  import javafx.scene.input.KeyEvent;
16  import javafx.scene.layout.HBox;
17  
18  /**
19   * This UI element is a container for elements related to changing the current
20   * playlist.
21   *
22   * @author ristolah
23   */
24  public class ChangePlaylistBox extends HBox implements Updateable {
25  
26      private TextField selector;
27  
28      /**
29       * Class constructor.
30       *
31       * Creates a new textfield, button for changing the playlist. Adds a
32       * KeyEvent listener for the textfield, if Enter is pressed the Change
33       * Playlist button's action will be performed.
34       *
35       * @param u Updateable which .update() will be called when playlist is
36       * changed.
37       * @param addTrackButton Already created AddTrackButton to add to this box.
38       */
39      public ChangePlaylistBox(Updateable u, AddTrackButton addTrackButton) {
40          super.setAlignment(Pos.CENTER);
41          super.alignmentProperty().isBound();
42          super.setSpacing(5);
43          selector = new TextField();
44          selector.setPromptText("Enter Playlist Name");
45          ChangePlaylistButton changePlaylistButton = new ChangePlaylistButton(selector, u);
46          selector.setOnKeyPressed(new EventHandler<KeyEvent>() {
47              @Override
48              public void handle(KeyEvent keyEvent) {
49                  if (keyEvent.getCode() == KeyCode.ENTER) {
50                      changePlaylistButton.changePlaylist(selector, u);
51                  }
52              }
53          });
54          ObservableList<Node> components = super.getChildren();
55          components.add(selector);
56          components.add(new ChangePlaylistButton(selector, u));
57          components.add(addTrackButton);
58      }
59  
60      @Override
61      public void update() {
62          selector.clear();
63          selector.setDisable(false);
64      }
65  
66  }
