4   import com.djrapitops.nmplayer.functionality.Track;

6   import javafx.event.ActionEvent;
7   import javafx.event.EventHandler;
8   import javafx.geometry.Pos;
9   import javafx.scene.control.Button;
10  
11  /**
12   * This button is used to change the track to the track represented by the
13   * UITrack element.
14   *
15   * @author ristolah
16   */
17  public class SelectButton extends Button {
18  
19      /**
20       * Constructor for the button. Sets the click event response to play the
21       * associated track from Playlist.
22       *
23       * Will check if the track is already playing, and does nothing if it is.
24       *
25       * @param uiTrack A UITrack Component to update when the button is pressed.
26       * @see MusicPlayer
27       * @see UITrack
28       * @see PlaylistManager
29       */
30      public SelectButton(UITrack uiTrack) {
31          final MusicPlayer mp = MusicPlayer.getInstance();
32          Track track = uiTrack.getTrack();
33          if (track.equals(mp.getCurrentTrack())) {
34              super.setStyle("-fx-background-color: #8290ed; -fx-text-fill: White");
35          } else {
36              super.setStyle("-fx-background-color: White");
37          }
38          super.setAlignment(Pos.CENTER_LEFT);
39          super.setText(uiTrack.getTrack().toString().replace("_", " "));
40          super.setPrefWidth(10000);
41          EventHandler h = (EventHandler<ActionEvent>) (ActionEvent event) -> {
42              if (!mp.getCurrentTrack().equals(uiTrack.getTrack()) || !mp.isPlaying()) {
43                  mp.selectTrack(track);
44                  mp.play();
45                  uiTrack.update();
46              }
47          };
48          super.setOnAction(h);
49      }
50  
51  }
