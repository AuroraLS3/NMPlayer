1   package com.djrapitops.nmplayer.ui.playlist;
2   
3   import com.djrapitops.nmplayer.functionality.Track;
4   import com.djrapitops.nmplayer.ui.Updateable;
5   import javafx.geometry.Pos;
6   import javafx.scene.layout.HBox;
7   
8   /**
9    * This element represents a Track object. Contains button to swap to this
10   * track, or remove it from the playlist.
11   *
12   * @author ristolah
13   */
14  public class UITrack extends HBox implements Updateable {
15  
16      private final Track track;
17      private final Updateable u;
18  
19      /**
20       * Constructor.
21       *
22       * Requires a Track object to get the text for the element and for removing
23       * it from the playlist. Updatable will be called if either of the buttons
24       * is pressed.
25       *
26       * @param track Track this element represents
27       * @param u Updateable to call when a button is pressed.
28       */
29      public UITrack(Track track, Updateable u) {
30          this.u = u;
31          this.track = track;
32          super.setAlignment(Pos.CENTER_LEFT);
33          super.alignmentProperty().isBound();
34          super.setSpacing(5);
35          super.setStyle("-fx-background-color: Lightgrey");
36          super.getChildren().add(new SelectButton(this));
37          super.getChildren().add(new RemoveButton(this));
38      }
39  
40      /**
41       * Used to get the Track object this element represents.
42       *
43       * @return the track.
44       */
45      public Track getTrack() {
46          return track;
47      }
48  
49      @Override
50      public void update() {
51          u.update();
52      }
53  }
