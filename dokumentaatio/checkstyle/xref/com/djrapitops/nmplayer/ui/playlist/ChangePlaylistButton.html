14  
15  /**
16   * JavaFx UI component, a Button used to change the current playlist.
17   *
18   * @author Rsl1122
19   * @see PlaylistFileManager
20   * @see PlaylistManager
21   * @see MusicPlayer
22   */
23  public class ChangePlaylistButton extends Button {
24  
25      /**
26       * Constructor for the button. Sets the click event response to change the
27       * playlist of MusicPlayer.
28       *
29       * @param t TextArea which text is used to read user input for playlist
30       * name.
31       * @param u A UI Component to update when the button is pressed.
32       * @see MusicPlayer
33       * @see UITrack
34       * @see PlaylistManager
35       */
36      public ChangePlaylistButton(TextField t, Updateable u) {
37          super.setText("Change Playlist");
38          super.setStyle("-fx-background-color: White");
39          super.setAlignment(Pos.CENTER_RIGHT);
40          EventHandler h = (EventHandler<ActionEvent>) (ActionEvent event) -> {
41              changePlaylist(t, u);
42          };
43          super.setOnAction(h);
44      }
45  
46      /**
47       * Changes the playlist according to the text inside the textfield.
48       *
49       * If the textfield is empty, MessageSender will be used to send the known
50       * playlists.
51       *
52       * @param t TextField used to get the playlist name.
53       * @param u Element to call .update() on after the change is complete. (If
54       * successful)
55       * @throws IllegalStateException If JavaFx Application is not running.
56       */
57      public void changePlaylist(TextField t, Updateable u) throws IllegalStateException {
58          String newPlaylist = t.getText().toLowerCase().trim();
59          if (newPlaylist.isEmpty()) {
60              MessageSender.getInstance().send(Phrase.EMPTY_NAME.parse(PlaylistFileManager.getKnownPlaylists()));
61              return;
62          }
63          MusicPlayer mp = MusicPlayer.getInstance();
64          mp.selectPlaylist(newPlaylist);
65          PlaylistManager playlistManager = mp.getPlaylistManager();
66          if (playlistManager.isRandom()) {
67              playlistManager.setRandom(true);
68          }
69          u.update();
70      }
71  
72  }
