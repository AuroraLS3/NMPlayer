2   package com.djrapitops.nmplayer.ui.toolbar;
3   
4   import com.djrapitops.nmplayer.functionality.MusicPlayer;
5   import com.djrapitops.nmplayer.ui.Updateable;
6   import javafx.event.ActionEvent;
7   import javafx.event.EventHandler;
8   import javafx.scene.control.Button;
9   
10  /**
11   * JavaFx UI component, a Button used to play/pause the currently playing track.
12   *
13   * @author Rsl1122
14   * @see MusicPlayer
15   */
16  public class PlayButton extends Button implements Updateable {
17  
18      /**
19       * Constructor for the Button.
20       * <p>
21       * Sets the click event response to either pause or play the playback with
22       * MusicPlayer and update the text on the button.
23       *
24       * @param u Updateable to call when the button is pressed.
25       * @see MusicPlayer
26       */
27      public PlayButton(Updateable u) {
28          super.setStyle("-fx-background-color: #8290ed; -fx-text-fill: White; -fx-font-weight: bold;");
29          super.setPrefWidth(40);
30          u.update();
31          EventHandler h = (EventHandler<ActionEvent>) (ActionEvent event) -> {
32              MusicPlayer musicPlayer = MusicPlayer.getInstance();
33              if (!musicPlayer.isPlaying()) {
34                  musicPlayer.play();
35              } else {
36                  musicPlayer.pause();
37              }
38              u.update();
39          };
40          super.setOnAction(h);
41      }
42  
43      @Override
44      public void update() {
45          if (!MusicPlayer.getInstance().isPlaying()) {
46              super.setText(" ▶ ");
47          } else {
48              super.setText("⏸");
49          }
50      }
51  
52  }
