1   package com.djrapitops.nmplayer.ui.toolbar;
2   
3   import com.djrapitops.nmplayer.functionality.MusicPlayer;
4   import com.djrapitops.nmplayer.ui.Updateable;
5   import javafx.beans.Observable;
6   import javafx.scene.control.Slider;
7   
8   /**
9    * This element is used to change the volume of the playback.
10   *
11   * @author Risto
12   * @see MusicPlayer
13   */
14  public class VolumeSlider extends Slider implements Updateable {
15  
16      VolumeSlider bar = this;
17  
18      /**
19       * Constructor.
20       *
21       * Will update the slider position to the current volume of the MusicPlayer.
22       * Adds a listener for user input that changes the volume of the playback.
23       */
24      public VolumeSlider() {
25          this.update();
26          super.valueProperty().addListener((Observable ov) -> {
27              if (bar.isValueChanging()) {
28                  MusicPlayer.getInstance().setVolume(bar.getValue() / 100);
29              }
30          });
31      }
32  
33      @Override
34      public void update() {
35          bar.adjustValue(MusicPlayer.getInstance().getVolume() * 100);
36      }
37  }
