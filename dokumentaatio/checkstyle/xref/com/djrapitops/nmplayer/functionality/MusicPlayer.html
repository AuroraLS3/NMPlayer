6   package com.djrapitops.nmplayer.functionality;
7   
8   import com.djrapitops.nmplayer.fileutils.PlaylistFileManager;
9   import com.djrapitops.nmplayer.fileutils.TrackFileManager;
10  import com.djrapitops.nmplayer.messaging.MessageSender;
11  import com.djrapitops.nmplayer.messaging.Phrase;
12  import com.djrapitops.nmplayer.functionality.playlist.PlaylistManager;
13  import java.io.File;
14  import javafx.scene.media.Media;
15  import javafx.scene.media.MediaPlayer;
16  
17  /**
18   *
19   * @author Risto
20   */
21  public class MusicPlayer {
22  
23      private final PlaylistManager playlist;
24      private final MessageSender msg;
25      private MediaPlayer mp;
26  
27      private Track currentTrack;
28      private int currentTrackIndex;
29      private String selectedPlaylist;
30      private boolean playing;
31  
32      public MusicPlayer() {
33          playlist = new PlaylistManager();
34          msg = MessageSender.getInstance();
35      }
36  
37      public void init() {
38          selectPlaylist("all");
39          playing = false;
40      }
41  
42      public void selectPlaylist(String playlistName) {
43          selectedPlaylist = playlistName;
44          playlist.setPlaylist(TrackFileManager.translateToTracks(PlaylistFileManager.load(selectedPlaylist)));
45          selectTrack(0);
46      }
47  
48      public void nextTrack() {
49          if (currentTrack != null) {
50              mp.stop();
51              playing = false;
52              selectTrack(currentTrackIndex + 1);
53              play();
54          }
55      }
56  
57      public void previousTrack() {
58          if (currentTrack != null) {
59              mp.stop();
60              playing = false;
61              selectTrack(currentTrackIndex - 1);
62              play();
63          }
64      }
65  
66      public void play() {
67          if (mp != null) {
68              playing = true;
69              mp.play();
70              msg.send(Phrase.NOW_PLAYING.parse(currentTrack.toString()));
71          }
72      }
73  
74      public void pause() {
75          if (mp != null && playing) {
76              playing = false;
77              mp.pause();
78              msg.send(Phrase.PAUSE + "");
79          }
80      }
81  
82      public void stop() {
83          if (mp != null && playing) {
84              playing = false;
85              mp.stop();
86              msg.send(Phrase.STOP + "");
87          }
88      }
89  
90      public void selectTrack(int i) {
91          final int tracks = playlist.getPlaylist().size();
92          if (playlist.isEmpty()) {
93              msg.send(Phrase.PLAYLIST_EMPTY + "");
94          } else if (i == -1) {
95              selectTrack(tracks - 1);
96          } else if (tracks > i && i >= 0) {
97              final String name = playlist.getPlaylist().get(i).getName();
98              selectTrack(name);
99          } else {
100             selectTrack(0);
101         }
102     }
103 
104     public void selectTrack(String trackName) {
105         Track track = playlist.getTrackByName(trackName);
106         if (track != null) {
107             String mp3FilePath = track.getFilePath();
108             File trackFile = new File(mp3FilePath);
109             if (!trackFile.exists()) {
110                 MessageSender.getInstance().send("File doesn't exist! " + track);
111                 return;
112             }
113             currentTrackIndex = playlist.getPlaylist().indexOf(track);
114             currentTrack = track;
115             msg.send(Phrase.SELECTED.parse(currentTrack.toString()));
116             Media play = new Media(trackFile.toURI().toString());
117             mp = new MediaPlayer(play);
118             mp.setOnEndOfMedia(new Runnable() {
119                 @Override
120                 public void run() {
121                     nextTrack();
122                 }
123             });
124         }
125     }
126 
127     public void addTrackToPlaylist(Track track) {
128         if (track == null) {
129             return;
130         }
131         playlist.addFilePathToPlaylist(track);
132         PlaylistFileManager.save(playlist.getPlaylist(), selectedPlaylist, true);
133         selectTrack(track.getName());
134     }
135 
136     public MediaPlayer getMediaPlayer() {
137         return mp;
138     }
139 
140     public boolean isPlaying() {
141         return playing;
142     }
143 
144     public static MusicPlayer getInstance() {
145         return MusicPlayerSingletonHolder.INSTANCE;
146     }
147 
148     private static class MusicPlayerSingletonHolder {
149 
150         private static final MusicPlayer INSTANCE = new MusicPlayer();
151     }
152 }
